107#F  ReplacedString( <string>, <old>, <new> )
108##
109##  This cannot be inside "kernel.g" because it is needed to read "kernel.g".
110##
111ReplacedString := function ( arg )
112    local str, substr, lss, subs, all, p, s, pp;
113    str := arg[1];
114    substr := arg[2];
115    lss := LEN_LIST( substr );
116    subs := arg[3];
117    if LEN_LIST( arg ) > 3  then
118        all := arg[4] = "all";
119    else
120        all := true;
121    fi;
122    p := POSITION_SUBSTRING( str, substr, 0 );
123    if p = fail  then
124        return str;
125    fi;
126    s := str{[  ]};
127    pp := 0;
128    while p <> fail  do
129        APPEND_LIST_INTR( s, str{[ pp+1 .. p - 1 ]} );
130        APPEND_LIST_INTR( s, subs );
131        pp := p + lss - 1;
132        if all  then
133            p := POSITION_SUBSTRING( str, substr, pp );
134        else
135            p := fail;
136        fi;
137        if p = fail  then
138            APPEND_LIST_INTR( s, str{[pp+1..LEN_LIST(str)]} );
139        fi;
140    od;
141    return s;
142end;

381##
382##  `ReadAndCheckFunc' creates a function that reads in a file named
