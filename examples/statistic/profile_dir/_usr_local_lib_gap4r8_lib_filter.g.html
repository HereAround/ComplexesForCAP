111#F  InstallTrueMethodNewFilter( <to>, <from> )
112##
113##  If <from> is a new filter then  it cannot occur in  the cache.  Therefore
114##  we do not flush the cache.  <from> should a basic  filter not an `and' of
115##  from. This should only be used in the file "type.g".
116##
117BIND_GLOBAL( "InstallTrueMethodNewFilter", function ( tofilt, from )
118    local   imp;
119
120    # Check that no filter implies `IsMutable'.
121    # (If this would be allowed then `Immutable' would be able
122    # to create paradoxical objects.)
123    if     IS_SUBSET_FLAGS( FLAGS_FILTER( tofilt ),
124                        FLAGS_FILTER( IS_MUTABLE_OBJ ) )
125       and not IS_IDENTICAL_OBJ( from, IS_MUTABLE_OBJ ) then
126      Error( "filter <from> must not imply `IsMutable'" );
127    fi;
128
129    imp := [];
130    imp[1] := FLAGS_FILTER( tofilt );
131    imp[2] := FLAGS_FILTER( from );
132    ADD_LIST( IMPLICATIONS, imp );
133    InstallHiddenTrueMethod( tofilt, from );
134end );

139#F  InstallTrueMethod( <newfil>, <filt> )
140##
141##  <#GAPDoc Label="InstallTrueMethod">
142##  <ManSection>
143##  <Func Name="InstallTrueMethod" Arg="newfil, filt"/>
144##
145##  <Description>
146##  It may happen that a filter <A>newfil</A> shall be implied by another
147##  filter <A>filt</A>, which is usually a meet of other properties,
148##  or the meet of some properties and some categories.
149##  Such a logical implication can be installed as an <Q>immediate method</Q>
150##  for <A>newfil</A> that requires <A>filt</A> and that always returns
151##  <K>true</K>.
152##  (This should not be mixed up with the methods installed via
153##  <Ref Func="InstallImmediateMethod"/>, which have to be called at runtime
154##  for the actual objects.)
155##  <P/>
156##  <Ref Func="InstallTrueMethod"/> has the effect that <A>newfil</A> becomes
157##  an implied filter of <A>filt</A>,
158##  see&nbsp;<Ref Sect="Filters"/>.
159##  <P/>
160##  For example, each cyclic group is abelian,
161##  each finite vector space is finite dimensional,
162##  and each division ring is integral.
163##  The first of these implications is installed as follows.
164##  <P/>
165##  <Log><![CDATA[
166##  InstallTrueMethod( IsCommutative, IsGroup and IsCyclic );
167##  ]]></Log>
168##  <P/>
169##  Contrary to the immediate methods installed with
170##  <Ref Func="InstallImmediateMethod"/>, logical implications cannot be
171##  switched off.
172##  This means that after the above implication has been installed,
173##  one can rely on the fact that every object in the filter
174##  <C>IsGroup and IsCyclic</C> will also be in the filter
175##  <Ref Func="IsCommutative"/>.
176##  </Description>
177##  </ManSection>
178##  <#/GAPDoc>
179##
180BIND_GLOBAL( "InstallTrueMethod", function ( tofilt, from )
181
182    InstallTrueMethodNewFilter( tofilt, from );
183
184    # clear the caches because we do not know if filter <from> is new
185    CLEAR_HIDDEN_IMP_CACHE( from );
186    CLEAR_IMP_CACHE();
187end );

192#F  NewFilter( <name>[, <implied>][, <rank>] )  . . . . . create a new filter
193##
194##  <#GAPDoc Label="NewFilter">
195##  <ManSection>
196##  <Func Name="NewFilter" Arg="name[, rank]"/>
197##
198##  <Description>
199##  <Ref Func="NewFilter"/> returns a simple filter with name <A>name</A>
200##  (see&nbsp;<Ref Sect="Other Filters"/>).
201##  The optional second argument <A>rank</A> denotes the incremental rank
202##  (see&nbsp;<Ref Sect="Filters"/>) of the filter,
203##  the default value is 1.
204##  <P/>
205##  The default value of the new simple filter for each object is
206##  <K>false</K>.
207##  </Description>
208##  </ManSection>
209##  <#/GAPDoc>
210##
211BIND_GLOBAL( "NewFilter", function( arg )
212    local   name,  implied,  rank,  filter;
213
214    if LEN_LIST( arg ) = 3  then
215      name    := arg[1];
216      implied := arg[2];
217      rank    := arg[3];
218    elif LEN_LIST( arg ) = 2  then
219      if IS_INT( arg[2] ) then
220        name    := arg[1];
221        implied := 0;
222        rank    := arg[2];
223      else
224        name    := arg[1];
225        implied := arg[2];
226        rank    := 1;
227      fi;
228    else
229      name    := arg[1];
230      implied := 0;
231      rank    := 1;
232    fi;
233
234    # Create the filter.
235    filter := NEW_FILTER( name );
236    if implied <> 0 then
237      InstallTrueMethodNewFilter( implied, filter );
238    fi;
239
240    # Do some administrational work.
241    FILTERS[ FLAG1_FILTER( filter ) ] := filter;
242    IMM_FLAGS:= AND_FLAGS( IMM_FLAGS, FLAGS_FILTER( filter ) );
243    RANK_FILTERS[ FLAG1_FILTER( filter ) ] := rank;
244    INFO_FILTERS[ FLAG1_FILTER( filter ) ] := 0;
245
246    # Return the filter.
247    return filter;
248end );
