323#F  RingByStructureConstants( <moduli>, <sctable> )
324#F  RingByStructureConstants( <moduli>, <sctable>, <name> )
325#F  RingByStructureConstants( <moduli>, <sctable>, <names> )
326#F  RingByStructureConstants( <moduli>, <sctable>, <name1>, <name2>, ... )
327##
328##  is an Z-module $M$ defined by the structure constants
329##  table <sctable> of length $n$, say.
330##
331##  The generators of $M$ are linearly independent abstract space generators
332##  $x_1, x_2, \ldots, x_n$ whose additive orders is given by the list
333##  <moduli>.  They are multiplied according to the formula
334##  $ x_i x_j = \sum_{k=1}^n c_{ijk} x_k$
335##  where `$c_{ijk}$ = <sctable>[i][j][1][i_k]'
336##  and `<sctable>[i][j][2][i_k] = k'.
337##
338InstallGlobalFunction( RingByStructureConstants, function( arg )
339    local T,      # structure constants table
340          n,      # dimensions of structure matrices
341          moduli, # additive orders of generators
342          zero,   # zero of `R'
343          names,  # names of the ring generators
344          Fam,    # the family of ring elements
345          A,      # the ring, result
346   filter,
347          gens;   # ring generators of `A'
348
349    # Check the argument list.
350    if not 1 < Length( arg ) and IsList( arg[1] ) 
351  and Length(arg[1])>0
352                                 and IsList( arg[2] ) then
353      Error( "usage: RingByStructureConstants([<moduli>,<sctable>]) or \n",
354             "RingByStructureConstants([<moduli>,<sctable>,<name1>,...])" );
355    fi;

358    n:=Length(moduli);

362    if   Length( arg ) = 2 then
363      names:= List( [ 1 .. n ],
364                    x -> Concatenation( "r.", String(x) ) );
365      MakeImmutable( names );
366    elif Length( arg ) = 3 and IsString( arg[3] ) then
367      names:= List( [ 1 .. n ],

387#X    # then all coefficients in this family are admissible.
388#X    # Otherwise only coefficients from `R' itself are allowed.
389#X    if Zero( ElementsFamily( FamilyObj( R ) ) ) <> fail then
390#X      SetFilterObj( Fam, IsFamilyOverFullCoefficientsFamily );
391#X    else

393#X    fi;

397    Fam!.names     := names;
398
399    # Construct the default type of the family.
400    Fam!.defaultTypeDenseCoeffVectorRep :=
401        NewType( Fam, IsSCRingObj and IsDenseCoeffVectorRep );
402
403    SetCoefficientsFamily( Fam, ElementsFamily( FamilyObj( Integers ) ) );
404    # temporary
405    SetIsUFDFamily(Fam,false);
406
407    # Make the generators and the ring.
408    SetZero( Fam, ObjByExtRep( Fam, List( [ 1 .. n ], x -> 0 ) ) );
409    gens:= Immutable( List( IdentityMat( n, Integers ),
410     x -> ObjByExtRep( Fam, x ) ) );
411    A:= RingByGenerators( gens );
412    SetIsWholeFamily(A,true);

416    elif Product(moduli)=0 then
417      SetSize(A,infinity);

436      elif moduli[1]=0 and n=-1 then
437 n:=ObjByExtRep(Fam,[-1]);
438 SetOne(Fam,n);
439 SetOne(A,n);
440      elif moduli[1]>1 and Gcd(n,moduli[1])=1 then
441 n:=1/n mod moduli[1];
442 n:=ObjByExtRep(Fam,[n]);
443 SetOne(Fam,n);
444 SetOne(A,n);
445      fi;
446    fi;
447
448    # Return the ring.
449    return A;
450end );
