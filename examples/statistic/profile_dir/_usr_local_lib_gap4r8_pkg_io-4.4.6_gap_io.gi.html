277InstallGlobalFunction( IO_ReadBlock, function( f, len )
278  # arguments: f ,len
279  # f must be an object of type IsFile
280  # len is the length to read
281  # Reads length bytes. Guarantees to return length bytes or less ""
282  # indicating EOF or fail for an error. Blocks until enough data arrives.

284  # before length bytes are read.

286  if not(IsFile(f)) or not(IsInt(len)) then
287      Error("Usage: IO_ReadBlock( f [,len] ) with IsFile(f) ",
288            "and IsInt(len)");
289  fi;
290  if f!.closed then
291      Error("Tried to read from closed file.");
292  fi;
293  res := "";
294  # First the case of no buffer:
295  if f!.rbufsize = false then
296      while Length(res) < len do
297          bytes := IO_read(f!.fd,res,Length(res),len - Length(res));
298          if bytes = fail then return fail; fi;
299          if bytes = 0 then return res; fi;   # this is EOF
300      od;
301      return res;
302  fi;
303  # read up to len bytes, using our buffer:
304  while Length(res) < len do

309          f!.rpos := f!.rpos + amount;

313          Append(res,f!.rbuf{[f!.rpos..f!.rpos+f!.rdata-1]});
314          f!.rpos := 1;
315          f!.rdata := 0;
316          if Length(res) = len then return res; fi;
317      fi;

319          return res;
320      fi;
321      if len - Length(res) > f!.rbufsize then
322          # In this case we read the whole thing:
323          bytes := IO_read(f!.fd,res,Length(res),len - Length(res));

325              return fail;
326          elif bytes = 0 then
327              return res;
328          fi;
329      else
330          # Now the buffer is empty, so refill it:
331          bytes := IO_read(f!.fd,f!.rbuf,0,f!.rbufsize);
332          if bytes = fail then
333              return fail;
334          elif bytes = 0 then
335              return res;
336          fi;
337          f!.rdata := bytes;
338      fi;
339  od;
